create or replace     function  crear_cat_estado_civil
(
	in nombre_ varchar(150),
	in descripcion_ text,
	in registrado_por_ int
)
returns table(
	num  integer,
	nombre varchar(150),
	descripcion text
)
    language 'plpgsql'
AS $body$

declare identificador integer = (select case when (select max(num_estado_civil) from cat_estado_civil) is null
			then 1 else (select max(num_estado_civil) from cat_estado_civil)+1 end);

begin

insert into cat_estado_civil 
(
	num_estado_civil ,
	nombre ,
	descripcion,
	registrado_por,
	fecha_registro 
)
values
(
	identificador,
	lower(nombre_),
	descripcion_,
	registrado_por_,
	current_timestamp
);
return query  select CE.num_estado_civil,
					 CE.nombre,
					 CE.descripcion 
					 from cat_estado_civil CE where CE.num_estado_civil = identificador limit 1;

end;
$body$;
