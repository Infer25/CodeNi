create or replace      function  crear_cat_tipo_colaborador
(
	in nombre_ varchar(150),
	in descripcion_ text,
	in registrado_por_ int
)
returns table(
	num  integer,
	nombre varchar(150),
	descripcion text
)
    language 'plpgsql'
AS $body$

declare identificador integer = (select case when (select max(num_tipo_colaborador) from cat_tipo_colaborador) is null
			then 1 else (select max(num_tipo_colaborador) from cat_tipo_colaborador)+1 end);

begin

insert into cat_tipo_colaborador 
(
	num_tipo_colaborador ,
	nombre ,
	descripcion,
	registrado_por,
	fecha_registro 
)
values
(
	identificador,
	lower(nombre_),
	descripcion_,
	registrado_por_,
	current_date
);
return query  select CE.num_tipo_colaborador,
					 CE.nombre,
					 CE.descripcion 
					 from cat_tipo_colaborador CE where CE.num_tipo_colaborador = identificador limit 1;
end;
$body$;
